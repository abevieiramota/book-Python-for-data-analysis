Anotando pontos¶

In [109]: 
t = 2*np.pi/3

fig = pl.figure(figsize=(10,8))
ax = fig.add_subplot(111)
ax.plot(X, C)
ax.plot(X, S)

# adiciona linhas conectando coordenada x ao ponto da curva do cosseno
ax.plot(np.ones(10)*t, np.linspace(cos(t), 0, 10), 'b--')
# anota
# xy : posição do ponto anotado
# xytext : posição da anotação
# arrowpropos : propriedades do arrow da anotação ao ponto
# ver: http://matplotlib.org/1.2.1/users/annotations_guide.html
ax.annotate(r'$\cos(\frac{2\pi}{3})=-\frac{1}{2}$', 
            xy=(t, np.cos(t)),
            xytext=(t+.2, np.cos(t)-.1),
            fontsize=16,
            arrowprops=dict(arrowstyle="->", connectionstyle="arc3,rad=.2"))
# adiciona linhas conectando coordenadas x ao ponto da curva do seno
ax.plot(np.ones(10)*t, np.linspace(0, sin(t), 10), 'g--')
# anota
ax.annotate(r'$\sin(\frac{2\pi}{3}=\frac{\sqrt{3}}{2}$',
            xy=(t, np.sin(t)),
            xytext=(t+.2, np.sin(t)+.2), 
            fontsize=16,
            arrowprops=dict(arrowstyle="->", connectionstyle="arc3, rad=.2"))

#             movendo spins
# deixa os spines do topo e da direita invisíveis
ax.spines['right'].set_color('none')
ax.spines['top'].set_color('none')
# move os spines do bottom e da esquerda
ax.spines['bottom'].set_position(('data', 0))
ax.spines['left'].set_position(('data', 0))

# remove os ticks no topo e à esquerda
ax.xaxis.set_ticks_position('bottom')
ax.yaxis.set_ticks_position('left')
